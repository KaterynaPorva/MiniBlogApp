212 public void ImportDataFromJson(string jsonContent)
213 {
214 _rwLock.EnterWriteLock();
215 try
216 {
217 // Читаємо прості DTO і збираємо з них повноцінні об'єкти
218 var importedDtos = JsonSerializer.Deserialize<List<PostBackupDto>>(jsonContent);
219 if (importedDtos != null)
220 {
221 foreach (var dto in importedDtos)
222 {
223 var existingPost = _posts.FirstOrDefault(p => p.Id == dto.Id);
224
225 if (existingPost == null)
226 {
227 // 1. Поста немає - Відновлюємо
228 var newPost = new Post
229 {
230 Id = dto.Id,
231 Title = dto.Title,
232 Content = dto.Content,
233 Author = dto.Author,
234 CreatedAt = dto.CreatedAt,
235 Likes = dto.Likes ?? new List<Like>(),
236 Comments = dto.Comments?.Select(c => new Comment
237 {
238 Id = c.Id,
239 Author = c.Author,
240 Text = c.Text,
241 Replies = new List<ICommentComponent>()
242 }).ToList() ?? new List<Comment>()
243 };
244
245 _posts.Add(newPost);
246 }
247 else
248 {
249 // 2. Пост існує - Зливаємо лайки та коментарі (Merge)
250 if (dto.Likes != null)
251 {
252 existingPost.Likes ??= new List<Like>();
253 var existingUsers = existingPost.Likes.Select(l => l.Username).ToHashSet();
254
255 foreach (var like in dto.Likes)
256 {
257 if (!existingUsers.Contains(like.Username))
258 {
259 existingPost.Likes.Add(like);
260 }
261 }
262 }
263
264 if (dto.Comments != null)
265 {
266 existingPost.Comments ??= new List<Comment>();
267 var existingCommentIds = existingPost.Comments.Select(c => c.Id).ToHashSet();
268
269 foreach (var cDto in dto.Comments)
270 {
271 if (!existingCommentIds.Contains(cDto.Id))
272 {
273 existingPost.Comments.Add(new Comment
274 {
275 Id = cDto.Id,
276 Author = cDto.Author,
277 Text = cDto.Text,
278 Replies = new List<ICommentComponent>()
279 });
280 }
281 }
282 }
283 }
284 }
285
286 // Оновлюємо лічильники
287 if (_posts.Any())
288 {
289 _nextPostId = _posts.Max(p => p.Id) + 1;
290 var allComments = _posts.Where(p => p.Comments != null).SelectMany(p => p.Comments);
291 if (allComments.Any())
292 {
293 _nextCommentId = allComments.Max(c => c.Id) + 1;
294 }
295 }
296 }
297 }
298 catch (JsonException ex)
299 {
300 throw new Exception("Не вдалося імпортувати дані. Перевірте цілісність JSON.", ex);
301 }
302 finally { _rwLock.ExitWriteLock(); }
303 }
